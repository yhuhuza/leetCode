Two sum
Given an array of integers nums and an integer target, return indices of the two numbers such that they add up to target. You may assume that each input would have exactly one solution, and you may not use the same element twice. You may assume that each input would have exactly one solution, and you may not use the same element twice. You can return the answer in any order. 

        var twoSum = function(nums, target) {

            const arrLen = nums.length;

            let mass = [];

            for (let i=0; i <= arrLen; i++) {

                const number = nums[i];

                for (let q=0; q <= arrLen; q++) {

                    const num = nums[q];

                    if (i === q) continue;

                    if ((number + num) === target && mass.length < 2) {

                        mass.push(i);

                        mass.push(q);

                    }

                }

            }

            return  mass;

        };
